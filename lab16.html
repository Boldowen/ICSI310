AA_1_TO_3 = {
    "A":"Ala","R":"Arg","N":"Asn","D":"Asp","C":"Cys","Q":"Gln","E":"Glu","G":"Gly",
    "H":"His","I":"Ile","L":"Leu","K":"Lys","M":"Met","F":"Phe","P":"Pro","S":"Ser",
    "T":"Thr","W":"Trp","Y":"Tyr","V":"Val","*":"Stop","?":"???"
}

STOP_CODONS_DNA = {"TAA", "TAG", "TGA"}
STOP_CODONS_RNA = {"UAA", "UAG", "UGA"}

def clean_seq(seq: str) -> str:
    return "".join(seq.split()).upper().replace("’", "'").replace("–", "-")

def dna_to_mrna(dna_5to3: str) -> str:
    return dna_5to3.replace("T", "U")

def translate_rna_from_first_aug(rna_5to3: str):
    start_idx = rna_5to3.find("AUG")
    if start_idx == -1:
        return [], start_idx, None 

    aa1 = []
    stop_codon = None
    for i in range(start_idx, len(rna_5to3) - 2, 3):
        codon = rna_5to3[i:i+3]
        aa = CODON_TABLE_RNA.get(codon, "?")
        aa1.append(aa)
        if aa == "*":
            stop_codon = codon
            break
    return aa1, start_idx, stop_codon
